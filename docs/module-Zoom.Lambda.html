import { compose } from '@dustinws/zoom/packages/core';

const square = n => n * n;

(static) memoize(value) → {function}

Source:
  • lambda/memoize.js, line 3
Since:
  • v2.0.0
Create a function that only runs for arguments it hasn't computed yet.
Example
// memoize :: (a -> b) -> (a -> b)
import { memoize } from '@dustinws/zoom/packages/core';

let count = 0;

function someExpensiveOperation() {
  count += 1;

  return count;
}

const memoized = memoize(someExpensiveOperation);

memoized(1); // 1
memoized(1); // 1
memoized(1); // 1
memoized(1); // 1

count; // 1

memoized(2); // 2
memoized(2); // 2
memoized(2); // 2
memoized(2); // 2

count; // 2
